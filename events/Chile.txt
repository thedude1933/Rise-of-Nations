#Meat Riots
country_event = {
	id = dnc_chile.1
	title = dnc_chile.1.t
	desc = dnc_chile.1.d
	trigger = {
		tag = CHL
		date > 1905.10.1
		date < 1905.11.1
	}
	fire_only_once = yes

	option = {
		name = dnc_chile.1.a
		set_variable = { CHL_meat_riot_pp_cost = -50 }
		add_political_power = -50
		random_list = {
			33 = {
				country_event = dnc_chile.2
			}
			66 = {
				country_event = dnc_chile.4
				subtract_from_variable = { CHL_meat_riot_pp_cost = 25 }
			}
		}
	}
	option = {
		name = dnc_chile.1.b
		add_stability = -0.1
		add_war_support = -0.1
		country_event = dnc_chile.3
	}
	
}
